## 2. Alert Hujan Lebat untuk Aktivitas Outdoor

**Konteks:** Penyelenggara event outdoor (festival, kegiatan olahraga, atau
acara komunitas) membutuhkan early warning hujan lebat untuk pengambilan
keputusan terkait pelaksanaan event.

**Solusi:** Notifikasi yang dikirim ketika prakiraan menunjukkan hujan lebat
(weather code ≥63 atau presipitasi ≥10mm) dalam 6 jam ke depan.

```python
import requests
from datetime import datetime, timedelta

ADM4 = "13.71.09.1001"  # Pasar Ambacang, Kuranji
HEAVY_RAIN_CODES = [63, 95, 97]  # Hujan Lebat, Hujan Petir

def check_heavy_rain():
    url = f"https://api.bmkg.go.id/publik/prakiraan-cuaca?adm4={ADM4}"
    headers = {"Accept": "application/json"}
    data = requests.get(url, headers=headers).json()

    # Prakiraan 6 jam ke depan (2 interval @ 3 jam)
    forecasts = data["data"][0]["cuaca"][0][:2]

    for f in forecasts:
        if f["weather"] in HEAVY_RAIN_CODES or f["tp"] >= 10:
            alert(f, data["lokasi"])

def alert(forecast, lokasi):
    msg = f"⛈️ Peringatan Hujan Lebat\n"
    msg += f"{lokasi['adm3']}, {lokasi['adm2']}\n"
    msg += f"{forecast['datetime']}: {forecast['weather_desc']}\n"
    msg += f"Curah: {forecast['tp']}mm, Angin: {forecast['ws']} km/h"
    # Send via Telegram / Slack / Email
    print(msg)

if __name__ == "__main__":
    check_heavy_rain()
```

function groupByDay(forecasts) {
	const days = {};
	for (const f of forecasts) {
		const day = f.local_datetime.split('T')[0]; // '2026-05-12'
		(days[day] ??= []).push(f);
	}
	return days;
}

function dailySummary(dayForecasts) {
	const temps = dayForecasts.map((f) => f.t);
	const codes = dayForecasts.map((f) => f.weather);
	const precip = dayForecasts.map((f) => f.tp);

	// Cuaca dominan (mode)
	const counts = {};
	codes.forEach((c) => (counts[c] = (counts[c] ?? 0) + 1));
	const dominantCode = Object.entries(counts).sort((a, b) => b[1] - a[1])[0][0];

	return {
		hi: Math.max(...temps),
		lo: Math.min(...temps),
		weather: parseInt(dominantCode),
		rainChance: precip.some((p) => p > 0) ? Math.max(...precip) : 0,
	};
}
```

function outdoorActivityScore(forecast) {
	const { t, weather, hu, ws } = forecast;

	// Hujan lebat/petir: tidak layak
	if ([63, 95, 97].includes(weather)) return 'Poor';
	// Hujan ringan-sedang
	if ([60, 61, 80].includes(weather)) return 'Caution';
	// Suhu tinggi + lembap = heat stress
	if (t >= 32 && hu >= 75) return 'Caution';
	// Angin kencang
	if (ws >= 25) return 'Caution';
	// Kondisi nyaman
	if (t >= 22 && t <= 28 && weather <= 3) return 'Excellent';
	return 'Good';
}
```
