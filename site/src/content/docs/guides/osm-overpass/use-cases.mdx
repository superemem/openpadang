```ts
// src/routes/api/poi/[category]/+server.ts
import { json } from '@sveltejs/kit';

export async function GET({ params }) {
	const { category } = params; // 'cafe', 'restaurant', dll

	const query = `
		[out:json][timeout:25];
		area["name"="Kota Padang"]["admin_level"="5"]->.padang;
		(
			node["amenity"="${category}"](area.padang);
			way["amenity"="${category}"](area.padang);
		);
		out center;
	`;

	const res = await fetch('https://overpass-api.de/api/interpreter', {
		method: 'POST',
		body: `data=${encodeURIComponent(query)}`,
		headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
	});

	const { elements } = await res.json();

	const pois = elements
		.filter((e) => e.tags?.name)
		.map((e) => ({
			id: e.id,
			name: e.tags.name,
			lat: e.lat ?? e.center?.lat,
			lng: e.lon ?? e.center?.lon,
			address: e.tags['addr:street'] ?? null,
		}));

	return json(pois);
}
```

```js
const query = `
	[out:json][timeout:25];
	area["name"="Kota Padang"]["admin_level"="5"]->.padang;
	(
		node["amenity"="restaurant"](area.padang);
		way["amenity"="restaurant"](area.padang);

async function getFoodByCategory() {
	const data = await overpassQuery(query);

	const grouped = {};
	for (const r of data.elements) {
		const cuisine = r.tags?.cuisine ?? 'unknown';
		const normalized = normalizeCuisine(cuisine);
		(grouped[normalized] ??= []).push({
			name: r.tags.name ?? 'Tanpa Nama',
			lat: r.lat ?? r.center?.lat,
			lng: r.lon ?? r.center?.lon,
			tags: r.tags,
		});
	}

	return grouped;
}

function normalizeCuisine(c) {
	const lower = c.toLowerCase();
	if (lower.includes('indonesian') || lower.includes('padang')) return 'Indonesian';
	if (lower.includes('chinese')) return 'Chinese';
	if (lower.includes('japanese') || lower.includes('korean')) return 'Japanese/Korean';
	if (lower.includes('burger') || lower.includes('pizza')) return 'Western/Fast Food';
	return 'Lainnya';
}
```

```js
async function findNearbyEmergency(lat, lng, radiusMeters = 5000) {
	const query = `
		[out:json][timeout:25];
		(
			node["amenity"="hospital"](around:${radiusMeters},${lat},${lng});
			node["amenity"="police"](around:${radiusMeters},${lat},${lng});
			node["amenity"="fire_station"](around:${radiusMeters},${lat},${lng});
		);
		out body;
	`;

	const data = await overpassQuery(query);

	return data.elements
		.map((e) => ({
			name: e.tags.name ?? 'Unnamed',
			type: e.tags.amenity,
			lat: e.lat,
			lng: e.lon,
			distance: haversine(lat, lng, e.lat, e.lon),
		}))
		.sort((a, b) => a.distance - b.distance);
}

function haversine(lat1, lng1, lat2, lng2) {
	const R = 6371; // Earth radius in km
	const dLat = ((lat2 - lat1) * Math.PI) / 180;
	const dLng = ((lng2 - lng1) * Math.PI) / 180;
	const a =
		Math.sin(dLat / 2) ** 2 +
		Math.cos((lat1 * Math.PI) / 180) *
			Math.cos((lat2 * Math.PI) / 180) *
			Math.sin(dLng / 2) ** 2;
	return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
}
```

```js
		node["historic"](area.padang);
		way["tourism"~"attraction|museum|viewpoint"](area.padang);
	);
	out center;
`;

async function getTourismLocations() {
	const data = await overpassQuery(query);

	return data.elements
		.filter((e) => e.tags?.name)
		.map((e) => ({
			name: e.tags.name,
			category: e.tags.tourism ?? e.tags.historic ?? 'unknown',
			lat: e.lat ?? e.center?.lat,
			lng: e.lon ?? e.center?.lon,
			description: e.tags.description ?? e.tags['description:id'] ?? null,
			wikipedia: e.tags.wikipedia ?? null,
		}));
}
```
